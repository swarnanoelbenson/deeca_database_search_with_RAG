## PROJECT BRIEF

You are building a **Natural Language Interface to Victoria's DataShare platform** for DEECA (Department of Energy, Environment and Climate Action).

**Goal:** Users ask for data in plain English. The system finds relevant DEECA datasets and explains what's available.

**Example interaction:**
```
User: "I need water quality data for the Murray region"
System: "Found 3 datasets:
1. Water Measurement Information System
   - Real-time water quality monitoring across Victoria
   - Owner: Water Group
   - Access: OFFICIAL (requires login)
   - Download: [link]

2. WMIS Historical Data (2010-2024)
   - Historical water quality records
   - Owner: DEECA
   - Access: PUBLIC
   - Download: [link]

Want more details on any of these?"
```

---

## ARCHITECTURE OVERVIEW

```
User Query (Gradio Chat Interface)
    ↓
Intent Parser (Claude API)
Extract: intent, data_type, region, filters
    ↓
Semantic Search (PostgreSQL + pgvector)
Find top 5 matching datasets
    ↓
RAG Synthesis (Claude API)
Generate natural language response with dataset details
    ↓
Gradio Display
Show results with download links
```

This is **RAG (Retrieval-Augmented Generation)** with semantic search.

**Frontend:** Gradio (chat interface, zero config)

import requests

CKAN_API_BASE = "https://discover.data.vic.gov.au/api/3/action"

def fetch_deeca_datasets() -> List[Dict]:
    """
    Fetch all DEECA datasets from Victoria's CKAN API.
    
    Returns:
        List of dataset dictionaries with metadata
    """
    url = f"{CKAN_API_BASE}/package_search"
    params = {
        "q": "organization:deeca",
        "rows": 200,
        "sort": "title_string asc"
    }
    
    try:
        response = requests.get(url, params=params, timeout=10)
        response.raise_for_status()
        
        data = response.json()
        if data['success']:
            return data['result']['results']
        else:
            print("CKAN API error:", data)
            return []
    
    except requests.exceptions.RequestException as e:
        print(f"Error fetching CKAN data: {e}")
        return []

def normalize_dataset(ckan_dataset: Dict) -> Dict:
    """
    Convert CKAN dataset format to our format.
    
    Args:
        ckan_dataset: Dataset from CKAN API
    
    Returns:
        Normalized dataset dictionary
    """
    return {
        'id': ckan_dataset.get('id', ''),
        'name': ckan_dataset.get('name', ''),
        'title': ckan_dataset.get('title', ''),
        'description': ckan_dataset.get('notes', ''),
        'owner': ckan_dataset.get('organization', {}).get('name', 'DEECA'),
        'category': [tag['name'] for tag in ckan_dataset.get('tags', [])],
        'last_updated': ckan_dataset.get('metadata_modified', ''),
        'access_level': 'PUBLIC',  # CKAN doesn't always have this; infer or default
        'formats': extract_formats(ckan_dataset.get('resources', [])),
        'url': ckan_dataset.get('url', ''),
        'api_endpoint': None,
        'coverage': ['Victoria'],  # Assume Victoria-wide
        'update_frequency': 'Unknown',
        'data_types': []
    }

def extract_formats(resources: List[Dict]) -> List[str]:
    """Extract file formats from CKAN resources"""
    formats = set()
    for resource in resources:
        if 'format' in resource:
            formats.add(resource['format'].upper())
    return list(formats) if formats else ['CSV', 'GeoJSON']

def populate_database():
    """
    Fetch datasets from CKAN API, generate embeddings, and populate database.
    """
    init_db()
    
    print("Fetching DEECA datasets from CKAN API...")
    ckan_datasets = fetch_deeca_datasets()
    
    if not ckan_datasets:
        print("No datasets found from CKAN API")

    print(f"Found {len(ckan_datasets)} datasets. Populating database...")
    
    session = get_session()
    
    for ckan_ds in ckan_datasets:

        # Parse last_updated date

            last_updated = datetime.fromisoformat(normalized['last_updated'].replace('Z', '+00:00'))

            last_updated = None

            title=normalized['title'],
            description=normalized['description'],
            owner=normalized['owner'],
            category=normalized['category'],

            api_endpoint=normalized['api_endpoint'],
            coverage=normalized['coverage'],
            update_frequency=normalized['update_frequency'],
            data_types=normalized['data_types'],
            description_embedding=embedding
        )
        
        session.merge(dataset)
    
    session.commit()
    session.close()
    
    print(f"✓ Populated database with {len(ckan_datasets)} datasets")

"""
Parse user queries to extract intent and entities.
Uses Claude for natural language understanding.
"""
from anthropic import Anthropic
import json
from typing import Dict
from src.config import MODEL

client = Anthropic()

class IntentParser:
    def __init__(self):
        self.model = MODEL
        self.conversation_history = []
    
    def parse_intent(self, user_query: str) -> Dict:
        """
        Parse user query to extract:
        - intent: what they're trying to do
        - data_type: what kind of data (biodiversity, water, fire, etc)
        - region: geographic area (Victoria-wide, specific region, etc)
        - constraints: access level, formats, update recency, etc
        
        Returns:
            Dictionary with structured intent information
        """
        
        system_prompt = """You are a data discovery assistant for DEECA.
Your job is to understand what environmental data users are looking for.

Extract from user queries:
1. Intent (find_datasets, explore_data, get_specific_dataset)
2. Data type (biodiversity, water, fire, energy, property, climate, etc)
3. Region/location (Victoria-wide, specific region, suburb, etc)
4. Constraints (access level needed, format preference, recency, etc)

Return ONLY a JSON object (no other text):
{
  "intent": "find_datasets",
  "data_type": ["biodiversity"],
  "region": ["Dandenongs"],
  "constraints": {
    "access_level": "PUBLIC",
    "format": ["CSV"],
    "update_recency_days": null
  },
  "clarification_needed": false,
  "clarification_question": null,
  "confidence": 0.95
}

If you need clarification, set clarification_needed=true."""

        # Add to conversation history
        self.conversation_history.append({
            "role": "user",
            "content": user_query
        })
        
        # Get Claude's response
        response = client.messages.create(
            model=self.model,
            max_tokens=500,
            system=system_prompt,
            messages=self.conversation_history
        )
        
        assistant_message = response.content[0].text
        self.conversation_history.append({
            "role": "assistant",
            "content": assistant_message
        })
        
        # Parse JSON from response
        try:
            json_start = assistant_message.find('{')
            json_end = assistant_message.rfind('}') + 1
            json_str = assistant_message[json_start:json_end]
            parsed = json.loads(json_str)
            return parsed
        except (json.JSONDecodeError, ValueError):
            # Fallback if JSON parsing fails
            return {
                "intent": "find_datasets",
                "data_type": [],
                "region": [],
                "constraints": {},
                "clarification_needed": True,
                "clarification_question": "Could you be more specific about what data you're looking for?",
                "confidence": 0.0
            }
    
    def reset(self):
        """Reset conversation history for next query"""
        self.conversation_history = []

#### Step 6: Create `semantic_search.py`

```python
"""
Search for datasets based on semantic similarity and filters.
"""
from src.database import get_session, Dataset
from src.embeddings import generate_embedding
from typing import List, Dict

class SemanticSearch:
    def search(
        self,
        query: str,
        intent: Dict,
        limit: int = 5
    ) -> List[Dict]:
        """
        Search for datasets using:
        1. Semantic similarity (embedding distance)
        2. Category/tag matching
        3. Region filtering
        4. Constraint matching
        
        Args:
            query: User's natural language query
            intent: Parsed intent from IntentParser
            limit: Max number of results to return
        
        Returns:
            List of matching datasets with metadata
        """
        
        session = get_session()
        
        # Generate embedding for query
        query_embedding = generate_embedding(query)
        
        # Start with base query
        base_query = session.query(Dataset)
        
        # Filter by data type/category if specified
        if intent.get('data_type'):
            data_types = intent['data_type']
            # Filter: dataset category contains ANY of the requested types
            base_query = base_query.filter(
                Dataset.category.overlap(data_types)
            )
        
        # Filter by region if specified
        if intent.get('region'):
            regions = intent['region']
            # Always include Victoria-wide datasets
            base_query = base_query.filter(
                Dataset.coverage.overlap(regions + ['Victoria'])
            )
        
        # Filter by constraints
        constraints = intent.get('constraints', {})
        
        if constraints.get('access_level'):
            access_level = constraints['access_level']
            base_query = base_query.filter(
                Dataset.access_level.in_([access_level, 'PUBLIC'])
            )
        
        # Get all candidates
        candidates = base_query.all()
        session.close()
        
        # Score candidates based on relevance
        scored = []
        for dataset in candidates:
            score = 0
            
            # Category match (3 points per match)
            if any(cat in dataset.category for cat in intent.get('data_type', [])):
                score += 3
            
            # Region match (2 points per match)
            if any(region in dataset.coverage for region in intent.get('region', [])):
                score += 2
            
            # Access level match (1 point)
            if constraints.get('access_level') and dataset.access_level == constraints['access_level']:
                score += 1
            
            # Format match (1 point)
            if constraints.get('format'):
                if any(fmt in dataset.formats for fmt in constraints['format']):
                    score += 1
            
            scored.append((dataset, score))
        
        # Sort by score descending, return top N
        scored.sort(key=lambda x: x[1], reverse=True)
        
        results = []
        for dataset, relevance_score in scored[:limit]:
            results.append({
                'id': dataset.id,
                'name': dataset.name,
                'title': dataset.title,
                'description': dataset.description,
                'owner': dataset.owner,
                'category': dataset.category,
                'last_updated': dataset.last_updated.isoformat() if dataset.last_updated else None,
                'access_level': dataset.access_level,
                'formats': dataset.formats,
                'url': dataset.url,
                'coverage': dataset.coverage,
                'update_frequency': dataset.update_frequency,
                'relevance_score': relevance_score
            })
        
        return results
```

"""
Generate natural language responses using retrieved datasets (RAG).
"""
from anthropic import Anthropic
import json
from typing import List, Dict
from src.config import MODEL

client = Anthropic()

class DatasetSynthesizer:
    def __init__(self):
        self.model = MODEL
    
    def synthesize_response(
        self,
        user_query: str,
        retrieved_datasets: List[Dict],
        intent: Dict
    ) -> str:
        """
        Use Claude to synthesize a natural language response
        based on retrieved datasets (RAG pattern).
        
        Args:
            user_query: Original user query
            retrieved_datasets: Top matching datasets from search
            intent: Parsed intent
        
        Returns:
            Natural language response with dataset recommendations
        """
        
        if not retrieved_datasets:
            return "I couldn't find datasets matching your request. Try being more specific about the region, data type, or access level needed."
        
        # Format retrieved datasets for context
        datasets_context = "Retrieved relevant datasets:\n\n"
        for i, ds in enumerate(retrieved_datasets, 1):
            datasets_context += f"{i}. **{ds['title']}**\n"
            datasets_context += f"   - Owner: {ds['owner']}\n"
            datasets_context += f"   - Description: {ds['description']}\n"
            datasets_context += f"   - Access Level: {ds['access_level']}\n"
            datasets_context += f"   - Available Formats: {', '.join(ds['formats'])}\n"
            datasets_context += f"   - Coverage: {', '.join(ds['coverage'])}\n"
            datasets_context += f"   - Last Updated: {ds['last_updated']}\n"
            datasets_context += f"   - URL: {ds['url']}\n\n"
        
        system_prompt = """You are a helpful data discovery assistant for DEECA (Victoria's Department of Energy, Environment and Climate Action).

Help users find and understand environmental datasets.

When presenting search results:
1. Explain which datasets match their query
2. Describe why each is relevant
3. Note any access restrictions
4. Highlight data freshness
5. Suggest next steps (how to access, format options, etc)

Be conversational, professional, and concise.
Keep explanations to 2-3 sentences per dataset."""

        prompt = f"""User Query: {user_query}

{datasets_context}

Provide a helpful response explaining these datasets and why they match the user's request.
Include download/access information."""

        response = client.messages.create(
            model=self.model,
            max_tokens=1200,
            system=system_prompt,
            messages=[
                {"role": "user", "content": prompt}
            ]
        )
        
        return response.content[0].text
    
    def format_with_links(
        self,
        response_text: str,
        datasets: List[Dict]
    ) -> str:
        """Add dataset links to the synthesized response"""
        
        formatted = response_text
        
        if datasets:
            formatted += "\n\n**Found Datasets:**\n"
            for ds in datasets:
                formatted += f"\n- [{ds['title']}]({ds['url']})\n"
                formatted += f"  *Owner: {ds['owner']} | Access: {ds['access_level']} | Updated: {ds['last_updated']}*\n"
        
        return formatted

"""
Gradio chat interface for DataShare NLP.
"""
import gradio as gr
from src.intent_parser import IntentParser
from src.semantic_search import SemanticSearch
from src.synthesizer import DatasetSynthesizer
from src.database import init_db

# Initialize components
intent_parser = IntentParser()
searcher = SemanticSearch()
synthesizer = DatasetSynthesizer()

def chat_response(message: str, history: list) -> str:
    """
    Multi-turn conversation handler.
    
    Flow:
    1. Parse intent from user message
    2. Check if clarification needed
    3. Search for relevant datasets
    4. Synthesize natural language response
    5. Format with links
    """
    
    # Step 1: Parse intent
    intent = intent_parser.parse_intent(message)
    
    # Step 2: Check for clarification
    if intent.get('clarification_needed'):
        response = intent['clarification_question']
        return response
    
    # Step 3: Search for datasets
    retrieved_datasets = searcher.search(message, intent, limit=5)
    
    # Step 4: Synthesize response
    response_text = synthesizer.synthesize_response(
        message,
        retrieved_datasets,
        intent
    )
    
    # Step 5: Format with links
    formatted_response = synthesizer.format_with_links(response_text, retrieved_datasets)
    
    return formatted_response

# Create Gradio interface
def create_interface():
    with gr.Blocks(
        title="DEECA Data Discovery",
        theme=gr.themes.Soft()
    ) as demo:
        gr.Markdown("""
        # 🌍 DEECA Data Discovery Assistant
        
        Ask for environmental data in plain English. I'll help you find DEECA's datasets.
        
        **Examples:**
        - "Find biodiversity data for the Dandenongs"
        - "Show me water quality datasets for the Murray region"
        - "What fire history data do we have for Victoria?"
        - "I need current water monitoring data"
        """)
        
        with gr.Row():
            with gr.Column():
                chatbot = gr.Chatbot(
                    label="Conversation",
                    height=500,
                    show_copy_button=True,
                    show_label=True
                )
                
                msg = gr.Textbox(
                    label="Ask for data",
                    placeholder="E.g., 'I need water quality data for Gippsland'",
                    lines=2,
                    scale=4
                )
                
                with gr.Row():
                    submit_btn = gr.Button("Search", scale=1, variant="primary")
                    clear_btn = gr.Button("Clear", scale=0)
        
        gr.Markdown("""
        ---
        **How it works:**
        1. Describe the data you're looking for in natural language
        2. The system parses your request and searches DEECA's dataset catalog
        3. You get relevant datasets with download links and details
        
        **Data categories:** Biodiversity • Water • Fire & Forests • Energy • Land & Property
        """)
        
        # Handle interactions
        def respond(message, chat_history):
            response = chat_response(message, chat_history)
            chat_history.append([message, response])
            return "", chat_history
        
        msg.submit(respond, [msg, chatbot], [msg, chatbot], queue=False)
        submit_btn.click(respond, [msg, chatbot], [msg, chatbot], queue=False)
        clear_btn.click(lambda: None, None, chatbot, queue=False)
    
    return demo

if __name__ == "__main__":
    # Initialize database
    print("Initializing database...")
    init_db()
    
    # Create and launch interface
    demo = create_interface()
    demo.launch(
        server_name="0.0.0.0",
        server_port=7860,
        share=False,
        show_api=False
    )
